489  LevelInd find_cell(T &particle, const LevelInd &cur_cell)
490  {
491  typename parallel::distributed::Triangulation<dim>::cell_iterator it, found_cell;
492  typename parallel::distributed::Triangulation<dim>::active_cell_iterator ait;
493  LevelInd res;
494 
495  // First check the last recorded cell since particles will generally stay in the same area
496  if (!triangulation_changed)
497  {
498  found_cell = typename parallel::distributed::Triangulation<dim>::cell_iterator(triangulation, cur_cell.first, cur_cell.second);
499  if (found_cell != triangulation->end() && found_cell->point_inside(particle.get_location()) && found_cell->active())
500  {
501  // If the cell is active, we're at the finest level of refinement and can finish
502  particle.set_local(found_cell->is_locally_owned());
503  return cur_cell;
504  }
505  }
506 
507  // Check all the cells on level 0 and recurse down
508  for (it=triangulation->begin(0); it!=triangulation->end(0); ++it)
509  {
510  res = recursive_find_cell(particle, std::make_pair(it->level(), it->index()));
511  if (res.first != -1 && res.second != -1) return res;
512  }
513 
514  // If we couldn't find it there, we need to check the active cells
515  // since the particle could be on a curved boundary not included in the
516  // coarse grid
517  for (ait=triangulation->begin_active(); ait!=triangulation->end(); ++ait)
518  {
519  if (ait->point_inside(particle.get_location()))
520  {
521  particle.set_local(ait->is_locally_owned());
522  return std::make_pair(ait->level(), ait->index());
523  }
524  }
525 
526  // If it failed all these tests, the particle is outside the mesh
527  particle.set_local(false);
528  return std::make_pair(-1, -1);

544  void send_recv_particles()
545  {
546  typename std::multimap<LevelInd, T>::iterator it;
547  typename parallel::distributed::Triangulation<dim>::cell_iterator found_cell;
548  int i;
549  unsigned int rank;
550  std::vector<T> send_particles;
551  typename std::vector<T>::const_iterator sit;
552 
553  // Go through the particles and take out those which need to be moved to another processor
554  for (it=particles.begin(); it!=particles.end();)
555  {
556  if (!it->second.local())
557  {
558  send_particles.push_back(it->second);
559  particles.erase(it++);
560  }
561  else
562  {
563  ++it;
564  }
565  }
566 
567  // Determine the total number of particles we will send to other processors
568  total_send = send_particles.size();
569  for (rank=0; rank<world_size; ++rank)
570  {
571  if (rank != self_rank) num_send[rank] = total_send;
572  else num_send[rank] = 0;
573  send_offset[rank] = 0;
574  }
575 
576  // Notify other processors how many particles we will be sending
577  MPI_Alltoall(num_send, 1, MPI_INT, num_recv, 1, MPI_INT, communicator);
578 
579  total_recv = 0;
580  for (rank=0; rank<world_size; ++rank)
581  {
582  recv_offset[rank] = total_recv;
583  total_recv += num_recv[rank];
584  }
585 
586  // Allocate space for sending and receiving particle data
587  unsigned int integrator_data_len = integrator->data_len();
588  unsigned int particle_data_len = T::data_len();
589  std::vector<double> send_data, recv_data;
590 
591  // Set up the space for the received particle data
592  recv_data.resize(total_recv*(integrator_data_len+particle_data_len));
593 
594  // Copy the particle data into the send array
595  for (i=0,sit=send_particles.begin(); sit!=send_particles.end(); ++sit,++i)
596  {
597  sit->write_data(send_data);
598  integrator->write_data(send_data, sit->get_id());
599  }
600 
601  // Exchange the particle data between domains
602  double *recv_data_ptr = &(recv_data[0]);
603  double *send_data_ptr = &(send_data[0]);
604  MPI_Alltoallv(send_data_ptr, num_send, send_offset, particle_type,
605  recv_data_ptr, num_recv, recv_offset, particle_type,
606  communicator);
607 
608  int put_in_domain = 0;
609  unsigned int pos = 0;
610  // Put the received particles into the domain if they are in the triangulation
611  for (i=0; i<total_recv; ++i)
612  {
613  T recv_particle;
614  LevelInd found_cell;
615  pos = recv_particle.read_data(recv_data, pos);
616  pos = integrator->read_data(recv_data, pos, recv_particle.get_id());
617  found_cell = find_cell(recv_particle, std::make_pair(-1,-1));
618  if (recv_particle.local())
619  {
620  put_in_domain++;
621  particles.insert(std::make_pair(found_cell, recv_particle));
622  }
623  }

634  void get_particle_velocities(const TrilinosWrappers::MPI::Vector &solution)
635  {
636  Vector<double> single_res(dim+2);
637  std::vector<Vector<double> > result;
638  Point<dim> velocity;
639  unsigned int i, num_cell_particles;
640  LevelInd cur_cell;
641  typename std::multimap<LevelInd, T>::iterator it, sit;
642  typename DoFHandler<dim>::active_cell_iterator found_cell;
643  std::vector<Point<dim> > particle_points;
644 
645  // Prepare the field function
646  Functions::FEFieldFunction<dim, DoFHandler<dim>, TrilinosWrappers::MPI::Vector> fe_value(*dof_handler, solution, *mapping);
647 
648  // Get the velocity for each cell at a time so we can take advantage of knowing the active cell
649  for (it=particles.begin(); it!=particles.end();)
650  {
651  // Save a pointer to the first particle in this cell
652  sit = it;
653 
654  // Get the current cell
655  cur_cell = it->first;
656 
657  // Resize the vectors to the number of particles in this cell
658  num_cell_particles = particles.count(cur_cell);
659  particle_points.resize(num_cell_particles);
660 
661  // Get a vector of the particle locations in this cell
662  i=0;
663  while (it != particles.end() && it->first == cur_cell)
664  {
665  if (it->second.vel_check()) particle_points[i++] = it->second.get_location();
666  it++;
667  }
668  result.resize(i, single_res);
669  particle_points.resize(i);
670 
671  // Get the cell the particle is in
672  found_cell = typename DoFHandler<dim>::active_cell_iterator(triangulation, cur_cell.first, cur_cell.second, dof_handler);
673 
674  // Interpolate the velocity field for each of the particles
675  fe_value.set_active_cell(found_cell);
676  fe_value.vector_value_list(particle_points, result);
677 
678  // Copy the resulting velocities to the appropriate vector
679  it = sit;
680  i = 0;
681  while (it != particles.end() && it->first == cur_cell)
682  {
683  if (it->second.vel_check())
684  {
685  for (int d=0; d<dim; ++d) velocity(d) = result[i](d);
686  it->second.set_velocity(velocity);
687  i++;
688  }
689  it++;
690  }
